```typescript
import { logger } from '../shared/logger.js';
import { AllProvidersExhaustedError, QueueTimeoutError, QueueFullError } from '../shared/errors.js';
import type { QueueItem, QueueStats } from './types.js';

export class RequestQueue {
  private queues = new Map<string, QueueItem[]>();

  constructor(private readonly maxSize: number = 100) {}

  enqueue(chainName: string, execute: () => Promise<unknown>, maxWaitMs: number): Promise<unknown> {
    // Check queue size for this chain
    const queue = this.queues.get(chainName) ?? [];
    if (queue.length >= this.maxSize) {
      return Promise.reject(new QueueFullError(chainName, this.maxSize));
    }

    return new Promise((resolve, reject) => {
      const item: QueueItem = {
        chainName,
        execute,
        resolve: (result) => { if (!item.settled) { item.settled = true; clearTimeout(item.timeoutHandle); resolve(result); } },
        reject: (error) => { if (!item.settled) { item.settled = true; clearTimeout(item.timeoutHandle); reject(error); } },
        enqueuedAt: Date.now(),
        timeoutHandle: setTimeout(() => {
          item.reject(new QueueTimeoutError(chainName, maxWaitMs));
          // Remove from queue
          this.removeItem(chainName, item);
        }, maxWaitMs),
        settled: false,
      };

      // Unref timeout so it doesn't block shutdown
      if (typeof item.timeoutHandle === 'object' && 'unref' in item.timeoutHandle) {
        item.timeoutHandle.unref();
      }

      queue.push(item);
      this.queues.set(chainName, queue);

      logger.debug({ chainName, depth: queue.length }, `Request queued for chain "${chainName}" (depth: ${queue.length})`);
    });
  }

  async drainOne(chainName: string): Promise<void> {
    const queue = this.queues.get(chainName);
    if (!queue || queue.length === 0) return;

    // Skip settled items (timed out or client disconnected)
    while (queue.length > 0 && queue[0].settled) {
      queue.shift();
    }
    if (queue.length === 0) return;

    const item = queue[0];

    try {
      const result = await item.execute();
      queue.shift();
      item.resolve(result);
      // Try next via microtask (not synchronous recursion)
      if (queue.length > 0) {
        queueMicrotask(() => this.drainOne(chainName));
      }
    } catch (error) {
      if (error instanceof AllProvidersExhaustedError) {
        // Still exhausted -- stop draining, item stays queued
        return;
      }
      // Non-rate-limit error -- fail this item, try next
      queue.shift();
      item.reject(error instanceof Error ? error : new Error(String(error)));
      if (queue.length > 0) {
        queueMicrotask(() => this.drainOne(chainName));
      }
    }
  }

  drainChains(chainNames: string[]): void {
    for (const name of chainNames) {
      this.drainOne(name);
    }
  }

  getStats(): QueueStats[] {
    const now = Date.now();
    return [...this.queues.entries()]
      .filter(([, q]) => q.length > 0)
      .map(([chainName, q]) => ({
        chainName,
        depth: q.length,
        oldestItemAgeMs: q[0] ? now - q[0].enqueuedAt : null,
      }));
  }

  rejectAll(error: Error): void {
    for (const [, queue] of this.queues) {
      for (const item of queue) {
        item.reject(error);
      }
    }
    this.queues.clear();
  }

  private removeItem(chainName: string, item: QueueItem): void {
    const queue = this.queues.get(chainName);
    if (!queue) return;
    const idx = queue.indexOf(item);
    if (idx !== -1) queue.splice(idx, 1);
  }
}
```
